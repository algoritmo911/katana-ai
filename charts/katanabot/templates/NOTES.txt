KatanaBot has been deployed.

1. Get the application URL by running these commands:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "katanabot.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Visit http://$NODE_IP:$NODE_PORT/health to check the application health."
{{- else if contains "LoadBalancer" .Values.service.type }}
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "katanabot.fullname" . }} --template "{{ range (index .status.loadBalancer.ingress 0) }}{{ . }}{{ end }}")
  echo "Visit http://$SERVICE_IP:{{ .Values.service.port }}/health to check the application health."
{{- else if contains "ClusterIP" .Values.service.type }}
  echo "The service is only reachable within the cluster."
  echo "To access it, run 'kubectl port-forward' and then visit http://127.0.0.1:8080/health"
  echo "kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "katanabot.fullname" . }} 8080:{{ .Values.service.port }}"
{{- end }}
